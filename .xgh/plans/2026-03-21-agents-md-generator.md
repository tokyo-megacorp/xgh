# AGENTS.md Generator Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the hand-maintained `AGENTS.md` with a generator that reads structured YAML config sources and produces an accurate, always-current document including an Automation Map derived from the trigger↔workflow↔agent binding.

**Architecture:** Five YAML config files (`project.yaml`, `team.yaml`, `workflow.yaml`, `triggers.yaml`, extended `agents.yaml`) serve as the single source of truth. A Python generator script (`scripts/gen-agents-md.sh`) reads all five plus `agents/*.md` frontmatter and emits `AGENTS.md`. Agent files gain `model`/`color`/`tools` frontmatter (mirroring lossless-claude's schema) so the roster section self-generates.

**Tech Stack:** Bash (`set -euo pipefail`), Python 3 + PyYAML, existing `assert_*` test helpers

**Spec:** `.xgh/specs/2026-03-21-agents-md-generator-design.md`

Create `scripts/gen-agents-md.sh`:

```bash
#!/usr/bin/env bash
# gen-agents-md.sh — Generate AGENTS.md from structured YAML config sources.
# Usage: bash scripts/gen-agents-md.sh
# Requires: python3, pyyaml (pip3 install pyyaml)

set -euo pipefail

REPO_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
cd "$REPO_ROOT"

# Check pyyaml
python3 -c "import yaml" 2>/dev/null || {
  echo "ERROR: pyyaml not found. Run: pip3 install pyyaml" >&2
  exit 1
}

python3 - <<'PYEOF'
import yaml, os, sys, glob, re
from datetime import date

ROOT = os.getcwd()

def load(path):
    with open(os.path.join(ROOT, path)) as f:
        return yaml.safe_load(f)

def frontmatter(path):
    """Extract YAML frontmatter from a markdown file."""
    with open(path) as f:
        content = f.read()
    m = re.match(r'^---\n(.*?)\n---', content, re.DOTALL)
    if not m:
        return {}
    return yaml.safe_load(m.group(1)) or {}

proj = load("config/project.yaml")
team = load("config/team.yaml")
wf   = load("config/workflow.yaml")
trg  = load("config/triggers.yaml")
ag   = load("config/agents.yaml")

# Load agent frontmatter from agents/*.md
agent_meta = {}
for path in sorted(glob.glob(os.path.join(ROOT, "agents", "*.md"))):
    fm = frontmatter(path)
    if "name" in fm:
        agent_meta[fm["name"]] = fm

# Load workflow names from config/workflows/*.yaml
workflows = {}
for path in sorted(glob.glob(os.path.join(ROOT, "config", "workflows", "*.yaml"))):
    with open(path) as f:
        w = yaml.safe_load(f)
    if w and "name" in w:
        workflows[w["name"]] = w

out = []

# Header
out.append(f"<!-- AUTO-GENERATED — do not edit. Run: bash scripts/gen-agents-md.sh -->")
out.append(f"")
out.append(f"# AGENTS.md — {proj['name']} ({proj['tagline']}) {proj.get('emoji','')}")
out.append(f"")
out.append(f"> Canonical agent instructions for AI systems working on the **{proj['name']}** repository.")
out.append(f"> All platform-specific files (CLAUDE.md, .github/copilot-instructions.md, etc.) point here.")
out.append(f"")
out.append(f"---")
out.append(f"")

# What is xgh
out.append(f"## What is {proj['name']}?")
out.append(f"")
out.append(proj['description'].strip())
out.append(f"")
out.append(f"Install via Claude Code plugin:")
out.append(f"")
out.append(f"```bash")
out.append(proj['install'].strip())
out.append(f"```")
out.append(f"")
out.append(f"---")
out.append(f"")

# Tech stack
out.append(f"## Tech Stack")
out.append(f"")
out.append(f"| Layer | Technology |")
out.append(f"|-------|-----------|")
for row in proj['tech_stack']:
    out.append(f"| {row['layer']} | {row['technology']} |")
out.append(f"")
out.append(f"---")
out.append(f"")

# Agent Roster — read from local_agents: section (not agents: which is platform registry)
out.append(f"## Agent Roster")
out.append(f"")
out.append(f"| Agent | Model | Capabilities |")
out.append(f"|-------|-------|-------------|")
for name, entry in ag.get('local_agents', {}).items():
    model = entry.get('model', agent_meta.get(name, {}).get('model', '—'))
    caps = ', '.join(entry.get('capabilities', []))
    out.append(f"| {name} | {model} | {caps} |")
out.append(f"")
out.append(f"---")
out.append(f"")

# Automation Map
out.append(f"## Automation Map")
out.append(f"")
out.append(f"Triggers, workflows, and agents — the full 'what fires when' picture.")
out.append(f"")
out.append(f"| Trigger | Condition | Skill | Workflow | Lead Agent |")
out.append(f"|---------|-----------|-------|----------|-----------|")
for t in trg.get('triggers', []):
    when = t['when']
    condition = f"{when.get('source','?')}: {when.get('event', when.get('type', when.get('path','?')))}"
    action = t.get('action', {})
    skill = action.get('skill', '—')
    wflow = action.get('workflow', '—')
    agent = action.get('agent', '—')
    out.append(f"| {t['name']} | {condition} | {skill} | {wflow} | {agent} |")
out.append(f"")
out.append(f"---")
out.append(f"")

# Development Guidelines
out.append(f"## Development Guidelines")
out.append(f"")
out.append(f"### General principles")
out.append(f"")
for i, c in enumerate(team['conventions']['general'], 1):
    out.append(f"{i}. {c}")
out.append(f"")
out.append(f"### Coding conventions")
out.append(f"")
